def self.account(organization_id, account_id)
  sql = %(
    SELECT journal_entries.date,
           journal_entries.journalable_type,
           journal_entries.journalable_id,
           journal_entry_lines.credit,
           journal_entry_lines.debit,
           accounts.classification,
           integrations.name
      FROM journal_entry_lines
INNER JOIN journal_entries ON journal_entries.id = journal_entry_lines.journal_entry_id
INNER JOIN accounts ON accounts.id = journal_entry_lines.account_id
INNER JOIN integrations ON integrations.id = journal_entries.integration_id
     WHERE accounts.organization_id = '#{organization_id}'
       AND accounts.id = '#{account_id}'
  ORDER BY journal_entries.date ASC
  )

  report = {
    lines: []
  }

  balance = 0

  ActiveRecord::Base.connection.exec_query(sql).rows.each do |row|
    balance += case row[5]
               when 'expense', 'asset'
                 row[4].to_f - row[3].to_f
               else
                 row[3].to_f - row[4].to_f
               end

    report[:lines] << {
      date: row[0],

      journalable_type: row[1],
      journalable_id: row[2],
      # journalable: row[1].constantize.find(row[2]),

      credit: row[3].to_f,
      debit: row[4].to_f,
      balance:,

      integration: row[6]
    }
  end

  report
end

def self.balance_sheet(organization_id)
  sql = %{
    SELECT *,
           CASE classification
           WHEN 'asset' THEN 1
           WHEN 'liability' THEN 2
           WHEN 'equity' THEN 3
           ELSE 999
           END AS rank
      FROM (SELECT accounts.classification AS classification,
                   accounts.id AS id,
                   accounts.reference AS reference,
                   accounts.name AS name,
                   CASE accounts.classification
                   WHEN 'asset' THEN SUM(debit) - SUM(credit)
                   WHEN 'liability' THEN SUM(credit) - SUM(debit)
                   WHEN 'equity' THEN SUM(credit) - SUM(debit)
                   END AS total,
                   accounts.internal_code
              FROM accounts
         LEFT JOIN journal_entry_lines ON accounts.id = journal_entry_lines.account_id
             WHERE accounts.organization_id = '#{organization_id}'
               AND accounts.classification IN ('asset', 'liability', 'equity')
               AND (accounts.internal_code IS NULL OR accounts.internal_code NOT IN ('CURRENT_YEAR_EARNINGS'))
          GROUP BY accounts.classification,
                   accounts.id,
                   accounts.reference,
                   accounts.name
             UNION ALL
            SELECT accounts.classification AS classification,
                   accounts.id AS id,
                   accounts.reference AS reference,
                   accounts.name AS name,
                   (SELECT SUM(credit) - SUM(debit) AS total
                      FROM accounts
                 LEFT JOIN journal_entry_lines ON accounts.id = journal_entry_lines.account_id
                     WHERE accounts.organization_id = '#{organization_id}'
                       AND accounts.classification IN ('income', 'expense')) AS total,
                   accounts.internal_code
              FROM accounts
             WHERE accounts.organization_id = '#{organization_id}'
               AND accounts.internal_code = 'CURRENT_YEAR_EARNINGS'
            ) AS SubQuery
      ORDER BY rank,
               reference
  }

  report = []
  last_classification = nil

  ActiveRecord::Base.connection.exec_query(sql).rows.each do |row|
    if last_classification != row[0]
      report << {
        name: row[0],
        total: 0,
        lines: []
      }
    end

    report[-1][:lines] << {
      account: {
        id: row[1],
        reference: row[2],
        name: row[3],
        internal_code: row[5]
      },
      total: row[4].to_f
    }

    report[-1][:total] += row[4].to_f

    last_classification = row[0]
  end

  report
end

def self.profit_and_loss(organization_id)
  acc = Organization.find(organization_id).accounts.find_by(internal_code: 'CURRENT_YEAR_EARNINGS')

  sql = %{
    SELECT *,
           CASE classification
           WHEN 'income' THEN 1
           WHEN 'expense' THEN 2
           ELSE 999
           END AS rank
      FROM (SELECT accounts.classification,
                   accounts.id,
                   accounts.reference,
                   accounts.name,
                   CASE accounts.classification
                   WHEN 'income' THEN SUM(credit) - SUM(debit)
                   WHEN 'expense' THEN SUM(debit) - SUM(credit)
                   END AS total
              FROM accounts
         LEFT JOIN journal_entry_lines ON accounts.id = journal_entry_lines.account_id
             WHERE accounts.organization_id = '#{organization_id}'
               AND accounts.classification IN ('income', 'expense')
          GROUP BY accounts.classification,
                   accounts.id,
                   accounts.reference,
                   accounts.name
             UNION ALL
            SELECT 'earnings' AS classification,
                   NULL AS id,
                   NULL AS reference,
                   NULL AS name,
                   SUM(credit) - SUM(debit) AS total
              FROM accounts
         LEFT JOIN journal_entry_lines ON accounts.id = journal_entry_lines.account_id
             WHERE accounts.organization_id = '#{organization_id}'
               AND accounts.classification IN ('income', 'expense')
            ) AS SubQuery
      ORDER BY rank,
               reference
  }

  report = []
  last_classification = nil

  ActiveRecord::Base.connection.exec_query(sql).rows.each do |row|
    if last_classification != row[0]
      report << {
        name: row[0],
        total: 0,
        lines: []
      }
    end

    report[-1][:lines] << {
      account: {
        id: row[0] == 'earnings' ? acc.id : row[1],
        reference: row[0] == 'earnings' ? acc.reference : row[2],
        name: row[0] == 'earnings' ? acc.name : row[3]
      },
      total: row[4].to_f
    }

    report[-1][:total] += row[4].to_f

    last_classification = row[0]
  end

  report
end

def self.sales_tax(organization_id)
  # Account type : SalesTax + Transfer (To SalesTax Account)

  # => 1) Starting Balance (As Of FROM_DATE)
  # => 2) Tax Amount On Sales During Period (CREDIT Between FROM_DATE And TO_DATE)
  # => 3) Tax Amount On Purchases During Period (DEBIT Between FROM_DATE And TO_DATE)
  # => 4) Net Tax Owing For Period (=2-3)
  # => 5) Less Payments To Governement During Period (Between FROM_DATE And TO_DATE)
  # => 6) Ending Balance (As Of TO_DATE)

  profit_and_loss(organization_id)
end
