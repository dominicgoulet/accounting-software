def accept_draft

        render partial: 'partials/flash'
      end
    else
      format.turbo_stream do
        flash.now.notice = 'Could not accept draft.'
        render partial: 'partials/flash'
      end
    end
  end
end

def create
  @commercial_document = current_organization.commercial_documents.build(document_params.merge(type: params[:type]))

  if @commercial_document.save
    respond_to do |format|
      format.turbo_stream do
        flash.now.notice = "#{params[:type]} created successfully."
        render partial: 'partials/flash'
      end
    end
  else
    render :new, status: :unprocessable_entity
  end
end

def destroy_many

      flash.now.notice = "#{@items.count(&:destroyed?)} #{document_name.pluralize} out of #{@items.size} were successfully deleted."
      render partial: 'partials/flash'
    end
  end
end

def send_email
  @email = current_organization.outgoing_emails.build(outgoing_email_params)
  @email.recipients = params[:outgoing_email][:recipients].split(',')
  @email.related_object = @invoice

  if @email.save
    @email.send!
    @invoice.mark_as_sent!

    respond_to do |format|
      format.turbo_stream do
        flash.now.notice = 'Invoice successfully sent by email.'
        render partial: 'partials/flash'
      end
    end
  else
    render :prepare_email, status: :unprocessable_entity
  end
end

def sti_path(action = nil)
  args = [params[:type].underscore.downcase.pluralize.to_sym]

  if action.present?
    args = if [:new].include? action
             [action, params[:type].underscore.downcase.to_sym]
           else
             [action, params[:type].underscore.downcase.pluralize.to_sym]
           end
  end

  polymorphic_path(args)
end
