def send_html_counter
  document_name = self.class.name.pluralize.downcase
  channel_name = "#{document_name}_status_filter".to_sym

  self.class.status.each_value do |status|
    broadcast_update_to [organization, channel_name],
                        target: "filters-#{document_name}-#{status}-count",
                        html: self.class.where(organization:, status:).size
  end

  broadcast_update_to [organization, channel_name],
                      target: "filters-#{document_name}-count",
                      html: self.class.where(organization:).size
end
