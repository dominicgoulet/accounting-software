def accept_draft
  respond_to do |format|
    eval <<-RUBY, binding, __FILE__, __LINE__ + 1
      @has_updated = @#{document_name}.accept_draft!
    RUBY

    if @has_updated
      format.turbo_stream do
        flash.now.notice = 'Draft accepted.'
        render partial: 'partials/flash'
      end
    else
      format.turbo_stream do
        flash.now.notice = 'Could not accept draft.'
        render partial: 'partials/flash'
      end
    end
  end
end

def create
  document = current_organization_documents.build(document_params)

  eval <<-RUBY, binding, __FILE__, __LINE__ + 1
    @#{document_name} = document
  RUBY

  if document.save
    respond_to do |format|
      format.turbo_stream do
        flash.now.notice = "#{document_name.camelize} created successfully."
        render partial: 'partials/flash'
      end
    end
  else
    render :new, status: :unprocessable_entity
  end
end

def destroy_many
  @items = current_organization_documents.where(id: params[:item_ids])
  @items.destroy_all

  respond_to do |format|
    format.turbo_stream do
      flash.now.notice = "#{@items.count(&:destroyed?)} #{document_name.pluralize} out of #{@items.size} were successfully deleted."
      render partial: 'partials/flash'
    end
  end
end

def index
  @status = params[:status]

  @q = current_organization_documents

  @q = @q.where(status: params[:status]) if @status.present?

  @q = @q.ransack(params[:q])

  eval <<-RUBY, binding, __FILE__, __LINE__ + 1
    @#{document_name.pluralize} = @q.result
  RUBY
end
